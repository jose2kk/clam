    use anyhow::Result;
    use owo_colors::{OwoColorize, Stream::Stdout};
    use crate::{config, state};

    pub fn execute() -> Result<()> {
        let cfg = config::load()?;
        let st = state::load()?;
        let active = st.active.as_deref();

        for profile in &cfg.profiles {
            if Some(profile.name.as_str()) == active {
                println!(
                    "* {}",
                    profile.name.if_supports_color(Stdout, |t| t.green().bold())
                );
            } else {
                println!("  {}", profile.name);
            }
        }

        Ok(())
    }
    ```

    Key: Active profiles get `* ` prefix with green+bold color (TTY only). Inactive get `  ` (two spaces) prefix. No header, no footer, no extra output.

    use anyhow::Result;
    use owo_colors::{OwoColorize, Stream::Stdout};
    use crate::{paths, state};

    pub fn execute() -> Result<()> {
        let st = state::load()?;

        let active = match st.active {
            Some(name) => name,
            None => {
                eprintln!("No active profile. Run `clmux add <name>` to create one.");
                std::process::exit(1);
            }
        };

        let dir = paths::profile_dir(&active)?;
        let dir_exists = dir.is_dir();

        let health = if dir_exists {
            "ok".if_supports_color(Stdout, |t| t.green()).to_string()
        } else {
            "missing".if_supports_color(Stdout, |t| t.red()).to_string()
        };

        println!("Profile: {}", active.if_supports_color(Stdout, |t| t.bold()).to_string());
        println!("Path:    {}", dir.display());
        println!("Status:  {}", health);

        Ok(())
    }
    ```
